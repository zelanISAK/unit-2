```

    i=0
    output = "| A | B | C |"
    A,B,C=False,False,False
    temp=1
    for i in range(8):
        output += '\n| ' + str(int(A)) + ' | ' + str(int(B)) + ' | ' + str(int(C)) + ' |'
        if temp>0 and temp%4==0:
            A=not A
        if temp>0 and temp%2==0:
```
