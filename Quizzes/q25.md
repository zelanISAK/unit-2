```.py
from matplotlib import pyplot as plt
import warnings
warnings.simplefilter('ignore')

x,y,s=[],[],-10.2
for i in range(100):
    s += 0.2
    x.append(s)
    temp = abs(s)
    y.append(temp)

plt.ylabel('$f(x) = |x|$')
plt.xlabel('x')
plt.plot(x,y)
plt.show()
```
