```
def get_l3tt3r(msg:str)->str:
    temp=[]
    switching = {'a':'4','e':'3','i':'1','o':'0',' ':'_'}
    for i in msg:
        if i in switching.keys():
            i=switching.get(i)
        temp+=i
    return ''.join(temp)
```
